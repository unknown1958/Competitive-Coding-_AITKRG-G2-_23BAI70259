#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        vector<int> a(n), b(n);
        for (int i = 0; i < n; i++) cin >> a[i];
        for (int i = 0; i < n; i++) cin >> b[i];
        int totalA = 0, totalB = 0;
        for (int i = 0; i < n; i++) { totalA += a[i]; totalB += b[i]; }
        vector<int> dp(totalA + 1, 0);
        dp[0] = 0;
        vector<bool> reachable(totalA + 1, false);
        reachable[0] = true;
        for (int i = 0; i < n; i++) {
            for (int j = totalA; j >= 0; j--) {
                if (reachable[j] && j + a[i] <= totalA) {
                    if (!reachable[j + a[i]] || dp[j + a[i]] < dp[j] + b[i]) {
                        reachable[j + a[i]] = true;
                        dp[j + a[i]] = dp[j] + b[i];
                    }
                }
            }
        }
        int ans = 0;
        for (int j = 0; j <= totalA; j++) {
            if (!reachable[j]) continue;
            int left = totalB - dp[j];
            int right = dp[j];
            int la = totalA - j, ra = j;
            if (la == 0 || ra == 0) { ans = max(ans, 0); continue; }
            ans = max(ans, min(left * ra, right * la));
        }
        cout << ans << "\n";
    }
    return 0;
}

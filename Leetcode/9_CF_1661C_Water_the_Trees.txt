#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        vector<long long> h(n);
        for (int i = 0; i < n; i++) cin >> h[i];
        long long mx = *max_element(h.begin(), h.end());
        map<long long, long long> cnt;
        for (int i = 0; i < n; i++) cnt[h[i]]++;
        auto cost = [&](long long target) -> long long {
            long long res = 0;
            for (auto& [val, c] : cnt) {
                if (val <= target) res += (target - val) * c;
            }
            return res;
        };
        long long ans1 = cost(mx);
        long long ans2 = cost(mx + 1);
        cout << min(ans1, ans2) << "\n";
    }
    return 0;
}
